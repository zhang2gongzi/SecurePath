    fseek(file, 0, SEEK_END);
    long length = ftell(file);
    if (length < 0) {
        fclose(file);
        return NULL;
    }
    rewind(file);

    char* buffer = (char*)malloc(length + 1);
    if (!buffer) {
        fclose(file);
        return NULL;
    }

    size_t bytes_read = fread(buffer, 1, length, file);
    if (bytes_read < (size_t)length) {
        if (ferror(file)) {
            free(buffer);
            fclose(file);
            return NULL;
        }
        length = (long)bytes_read;
    }
    buffer[length] = '\0';

    fclose(file);
    return buffer;
}

    }

    int written;
    int ret = snprintf(buf, bufsz, "%s [%s] ", time_str, level);

        va_list ap;

    }

    size_t remaining = bufsz - written;
    va_list ap;
    va_start(ap, fmt);

    va_end(ap);

    if (msg_written < 0) return msg_written;

    if ((size_t)(written + msg_written + 1) < bufsz) {
        buf[written + msg_written] = '\n';
        buf[written + msg_written + 1] = '\0';
        msg_written++;
    }

    return written + msg_written;
